    <script>
        (function() {
            const form = document.getElementById('searchForm');
            const searchInput = document.getElementById('searchInput');

            // 固定为百度，移除所有搜索引擎切换逻辑
            form.action = 'https://www.baidu.com/s';
            searchInput.name = 'wd';

            // 表单提交空值校验
            form.addEventListener('submit', function(e) {
                const query = searchInput.value.trim();
                if (query === '') {
                    e.preventDefault();
                    searchInput.focus();
                    searchInput.style.border = '1px solid #d14545';
                    setTimeout(() => { searchInput.style.border = 'none'; }, 800);
                }
            });

            // 壁纸切换
            const brushBtn = document.getElementById('brushBtn');
            const wallpaperPanel = document.getElementById('wallpaperPanel');
            const wallpaperItems = document.querySelectorAll('.wallpaper-item');

            const wallpaperMap = {
                'default': 'var(--bg-color)',
                'gray': '#e0e0e0',
                'star': 'linear-gradient(135deg, #0b1e33 0%, #2c3e50 100%)',
                'ocean': 'linear-gradient(135deg, #0077b6 0%, #90e0ef 100%)',
                'forest': 'linear-gradient(135deg, #2d6a4f 0%, #52b788 100%)',
                'aurora': 'linear-gradient(135deg, #1a1a2e 0%, #16213e 50%, #0f3460 100%)'
            };

            let currentWallpaper = localStorage.getItem('myWallpaper') || 'default';

            function applyWallpaper(key) {
                const bg = wallpaperMap[key];
                if (!bg) return;

                const body = document.body;
                if (key === 'default') {
                    body.classList.remove('custom-wallpaper');
                    body.style.background = '';
                    body.style.backgroundImage = '';
                } else {
                    body.classList.add('custom-wallpaper');
                    body.style.background = bg;
                    body.style.backgroundSize = 'cover';
                    body.style.backgroundPosition = 'center';
                    body.style.backgroundRepeat = 'no-repeat';
                    body.style.backgroundAttachment = 'fixed';
                }

                wallpaperItems.forEach(el => el.classList.remove('active'));
                const target = document.querySelector('.wallpaper-item[data-wallpaper="' + key + '"]');
                if (target) target.classList.add('active');

                localStorage.setItem('myWallpaper', key);
                currentWallpaper = key;
            }

            brushBtn.addEventListener('click', function(e) {
                e.stopPropagation();
                wallpaperPanel.classList.toggle('active');
            });

            document.addEventListener('click', function(e) {
                if (!wallpaperPanel.contains(e.target) && e.target !== brushBtn) {
                    wallpaperPanel.classList.remove('active');
                }
            });

            wallpaperItems.forEach(item => {
                item.addEventListener('click', function() {
                    const key = this.getAttribute('data-wallpaper');
                    applyWallpaper(key);
                    wallpaperPanel.classList.remove('active');
                });
            });

            applyWallpaper(currentWallpaper);

            // 导航管理 (右键修改/删除 + 添加)
            const navGrid = document.getElementById('navGrid');
            const contextMenu = document.getElementById('contextMenu');
            const editBtn = document.getElementById('editNavBtn');
            const deleteBtn = document.getElementById('deleteNavBtn');
            const addBtn = document.getElementById('addNavBtn');

            let currentTarget = null;

            function hideContextMenu() {
                contextMenu.classList.remove('active');
                currentTarget = null;
            }

            function showContextMenu(e, target) {
                e.preventDefault();
                currentTarget = target;
                contextMenu.style.left = Math.min(e.clientX, window.innerWidth - 140) + 'px';
                contextMenu.style.top = Math.min(e.clientY, window.innerHeight - 80) + 'px';
                contextMenu.classList.add('active');
            }

            function attachContextMenu(el) {
                el.addEventListener('contextmenu', function(e) {
                    showContextMenu(e, el);
                });
            }

            document.querySelectorAll('.nav-item').forEach(attachContextMenu);

            document.addEventListener('click', function(e) {
                if (!contextMenu.contains(e.target)) {
                    hideContextMenu();
                }
            });

            editBtn.addEventListener('click', function() {
                if (!currentTarget) return;
                const currentLabel = currentTarget.getAttribute('data-label') || currentTarget.querySelector('.label')?.textContent || '未命名';
                const currentHref = currentTarget.href || '';
                const newLabel = prompt('请输入新的导航名称：', currentLabel);
                if (newLabel === null) { hideContextMenu(); return; }
                const newHref = prompt('请输入新的网址 (以 http:// 或 https:// 开头)：', currentHref);
                if (newHref === null) { hideContextMenu(); return; }
                if (newLabel.trim() !== '' && newHref.trim() !== '') {
                    const labelEl = currentTarget.querySelector('.label');
                    if (labelEl) labelEl.textContent = newLabel.trim();
                    currentTarget.setAttribute('data-label', newLabel.trim());
                    currentTarget.href = newHref.trim();
                    const img = currentTarget.querySelector('.favicon');
                    if (img) {
                        try {
                            const domain = new URL(newHref.trim()).origin;
                            img.src = domain + '/favicon.ico';
                            img.style.display = 'block';
                            img.onerror = function() { this.style.display = 'none'; };
                        } catch(e) {
                            img.style.display = 'none';
                        }
                    }
                }
                hideContextMenu();
            });

            deleteBtn.addEventListener('click', function() {
                if (!currentTarget) return;
                if (confirm('确定要删除这个导航项吗？')) {
                    currentTarget.remove();
                }
                hideContextMenu();
            });

            addBtn.addEventListener('click', function() {
                const label = prompt('请输入导航名称：');
                if (label === null || label.trim() === '') return;
                const url = prompt('请输入网址 (以 http:// 或 https:// 开头)：');
                if (url === null || url.trim() === '') return;
                if (!url.trim().startsWith('http://') && !url.trim().startsWith('https://')) {
                    alert('请输入有效的网址，需以 http:// 或 https:// 开头');
                    return;
                }
                const newItem = document.createElement('a');
                newItem.className = 'nav-item';
                newItem.href = url.trim();
                newItem.target = '_blank';
                newItem.setAttribute('data-label', label.trim());
                let domain = '';
                try {
                    domain = new URL(url.trim()).origin;
                } catch(e) {
                    domain = '';
                }
                newItem.innerHTML = '<img class="favicon" src="' + (domain ? domain + '/favicon.ico' : '') + '" alt="" onerror="this.style.display=\'none\'" /><span class="label">' + label.trim() + '</span>';
                navGrid.appendChild(newItem);
                attachContextMenu(newItem);
            });

            console.log('极简搜索 · 百度专用 (可定制导航/壁纸)');
        })();
    </script>
